1. Дан код (объяснить результаты):
var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 - В префиксной форме инкрементирование производится сразу, а возврат — уже с обновленным значением. Т.е. а сначала увеличилось на 1, а потом записаловь в c.
d = b++; alert(d); // 1 - В постфиксной форме сначала происходит возвращение значения, а потом выполняется инкрементирование. Т.е. сначала произошла запись b в d (1), а только потом b увеличилось на 1
c = (2+ ++a); alert(c); // 5 - a уже увеличлось выше на 1 (стало 2), и тут еще увеличилось на 1 (стало 3), 2 + 3 = 5
d = (2+ b++); alert(d); // 4 - b выше увеличилось на 1 (стало 2), а тут сначала происходит сложение с 2 (2 + 2 = 4) и присвоение в d, а только потом увеличение b еще на 1 (стало 3)
alert(a); // 3 - описано выше
alert(b); // 3 - описано выше

7. *Сравнить null и 0. Попробуйте объяснить результат.
null && 0 = null - Объединение, но нельзя объединить с ничем (null), по-этому возвращает null.
null || 0 = 0 - пересечение (или то, или это), по-этому возвразает 0
null > (<, ==) 0 = false - при сравлении с ничем всегда будет false.
